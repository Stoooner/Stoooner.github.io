Python正则表达式

python中正则表达式的使用方法总结.

import re

# 1.表示字符和表示数量

#使用match方法进行匹配: result = re.match(正则表达式，要匹配的字符串)
pattern = "itcast"
s = "1itcastitheima"
# result = re.match('\w\W',s) #从左到右进行匹配，如果一开始就规则与内容不一致则停止工作,不再往下进行匹配
# print(result.group())# group()方法返回匹配的部分

#[^345678]的^号代表对方括号中345678的取反(当然，也可以使用[^3^4^5^6^7^8]，两者等价)，也就是除了345678之外的任意一个都满足
result1 = re.match('1[^345678]','19') #这里代表第二位是除了345678之外的任意
result2 = re.match('1[a-z5-9]','17') #代表第二位是a-z或者5-9的字符
#\d == [0-9]; \D == [^0-9]; \w == [a-zA-Z0-9_]; \W == [^a-zA-Z0-9_];也就是说方括号的方式和等号前的方式是等价的
result3 = re.match('\d*','12abc') # \d*代表匹配0个或者多个数字，等价于None或者\d\d\d\d\d....，而不是代表匹配多个同样的数字出现多次
result4 = re.match('\d+','132xsx') # \d+代表匹配1至少1个或者或者多个数字
result5 = re.match('\d?','12abc') # \d?代表匹配0次数字或者1次数字，即要么1次，要么没有，多个数字也最多匹配到第一个数字
result6 = re.match('\d*[a-z]','123abc') # \d* 或者\d+也都可以，只要不是\d?就能匹配"123abc"
result7 = re.match('\d{3}[a-z]','1234acbss') # \d{3}[a-z]代表匹配前面的数字要出现3次(等价于\d\d\d[a-z])，后面是a-z的字符.
result8 = re.match('\d{3,}[a-z]','12776xasx') # \d{3,}[a-z]代表前面数字匹配的个数是至少要有3个(>=3)，其中\d{1,} == \d+; \d{0,} == \d*
result9 = re.match('\d{3,5}[a-z]','1234xasx') # \d{3,5}代表数字出现3次到5次之间都行
result10 = re.match('1[35678]\d{9}','17289890781xasx') # 匹配电话号码(暂时不考虑边界)
print(result10.group())


# 2.原始字符串

s1 = r"\\nabc" #假使我们的例子中某个字符串就是\nabc，那么为了避免解释器将其中的\n转义为换行，则需要再加上一个\
s2 = r'\nabc' #加上r可以避免\n被转义，其中r是英文字母raw的缩写，代表原始格式(也就是输入字符串是什么样子的，输出的时候就是什么样子的)
# 其中s1和s2的表达式一样的（通常在写正则的时候，规则处都要加上r，避免一些内容被转义掉）
result = re.match(r"\\nabc",s2)
print(result.group())


# 3.表示边界(需要注意的是，^符号在[]中代表取反，但是不在[]中的时候代表以某个部分开头，而$代表以某个部分结尾)

#需要注意的是在Ubuntu下的vim中A表示光标进入尾部且进入编辑模式，I表示进入头部且进入编辑模式；
# 若不进入编辑模式，则shift+4($)就直接让光标进入尾部，相应的shift+6(^)则在不进入编辑模式的情况下进入头部。
result1 = re.match(r"1[35678]\d{9}$","15198168018").group() # r"1[35678]\d{9}$"代表匹配电话号码，且以9个数字结尾,多1位都无法匹配
result2 = re.match(r"^P\d+EIVEL\b",'P080978EIVEL').group() # r"^P\d+EIVEL"代表匹配以P开头，一个或者多个数字在中间(\d+)，并紧接着出现EIVEL作为结尾(\b)的内容
result3 = re.match(r"^P\d+\s\bEIVEL\b",'P080978 EIVEL RS').group() # \b这些的只是包含边界信息，不包含字符
result4 = re.match(r"^P.+\bEIVEL\b",'P080978 EIVEL RS').group() # result3和resul4实质都是一样的，注意.代表匹配任意1个字符（除了\n），.+就是匹配任意多个字符
result5 = re.match(r"^P.+EIVEL\d{4}",'P080978EIVEL0000').group() #匹配开头是P,中间EIVEL的内容，其中不能以EIVEL结尾(也就是EIVEL后面得有字符)。
print(result5)


# 4.匹配分组

# | 用于匹配左右两边任意一个表达式:
result = re.match(r"[1-9]?\d?$|100$",'0').group() #匹配0-100之间的数字
#(.*)代表匹配0个或者多个内容，而group(1)中的1代表正则表达式中出现的第一组括号的内容，如果为2则是第二组括号的内容
# 需要注意的是group()中是不传值的时候等价于group(0)，也就是给出全部匹配的内容
result2 = re.match(r"<h1>(.*)</h1>","<h1>匹配分组</h1>").group(1)
# groups()代表一次性将多个括号的内容以元祖的方式呈现出现:('<h1>', '</h1>'),其中groups()[0] == group(1)
result3 = re.match(r"(<h1>).*(</h1>)","<h1>匹配分组</h1>").groups()
s = "<html><h1>itcast</h1></html>"
#如果需要校验拿到的文本是否格式正确的时候，需要对于重要的内容使用()给括起来
#而且\num代表拿到第num个()所拿到的内容，所以\2所在的尖括号的内容得和第二个括号的内容一致，相应的，\1所在尖括号的内容得和第一个括号的一致
result4 = re.match(r"<(.+)><(.+)>.+</\2></\1>",s)

#匹配邮箱账号
# (\w+)代表邮箱号前面的账号，需要被我们提取出来，所以加上()
# (这里使用括号把163|126|gmail|qq给括起来是因为，
# 这个时候163|126|gmail|qq和前面的内容不再是各自独立的某个情况了，(前面匹配0到100的例子中|左右各自都是一种独立的情况，而这里不是)
# 这里163|126|gmail|qq仅仅是整个正则表达式的某个部分而已，所以需要使用括号将其作为一个分组，
# 而在这个分组里这4种情况才是各自独立的情况)(这里\.是因为.不是正则表达式中的标识符，而仅仅是一个邮箱号的某个组成部分，所以需要加上\进行转义)
result5 = re.match(r"(\w+)@(163|126|gmail|qq)\.(com|cn|net)$",'1786957287@qq.com')

#给用于提取内容的括号分组起名字
#需要注意的是，如果在某个正则表达式中()分组特别多的时候，为了避免混淆，这个时候给各个括号分组取不同的名字，例如：
#在<(?P<name>.+)>中(?P<name>)实际上是不属于正则表达式的部分的，而是用于这次括号分组的名字的信息
#而在取用对应的括号分组的时候，使用(?P=name)的方式就能取用对应的括号分组
result6 = re.match(r"<(?P<key1>.+)><(?P<key2>.+)>.+</(?P=key2)></(?P=key1)>",s)
# print(result6.groups())


# 5.re模块的高级用法(search,findall)
s = "<html><h1>itcast</h1></html>"
# search方法代表的是从整个字符串s中从左向右开始去找，如果遇到第一个<不是要查找的itcast中的i,
# 则继续沿着第一个<往下走，直到匹配了正则表达式为止。
result = re.search(r"^itcast$",s) #这里加上^和$代表必须要以i开头，以t结尾的情况才能行。
# 需要注意的是search从左向右找到了第一个匹配的内容后，search就结束了.
s2 = "itcast</h1></html>itheima</h1>"
result2 = re.search(r"\w+</h1>",s2)
# 相应的如果你要把所有符合匹配条件的内容都找到，则需要使用findall,且对于的result3没有.group()方法，result3此时就是一个列表类型了
result3 = re.findall(r"\w+</h1>",s2)
# print(result3)
# 在vim中需要批量替换的时候方法是:  :row_num1,row_num2s/替换前的词/替换后的词/g   其中g代表替换全部
# 在正则表达式中使用sub进行批量替换(需要替换的部分,替换的内容(可以是一个函数),需要处理的整个内容)
result4 = re.sub(r"php","python","itcast python cpp php python php")
# print(result4)
# 需要注意的是在python中真正的注释是#号后面的内容，当解释器经过#的注释时直接略过，注释的内容也不会对内容产生影响
# 而"""xxxxxx"""并非真正的注释，只不过可以当成注释来使用而已，三引号只不过是字符串的一种特殊形式而已，可以将三引号之间
# 字符串的格式给保留下来，实际在使用的时候是会对内容有影响的，如果三引号中间的内容过大一样会引起程序的崩溃，这点是需要和真正的#注释相区分的。
# sub例题：
s3 = """
<div>
        <p>岗位职责：</p>
<p>完成推荐算法、数据统计、接口、后台等服务器端相关工作</p>
<p><br></p>
<p>必备要求：</p>
<p>良好的自我驱动力和职业素养，工作积极主动、结果导向</p>
<p>&nbsp;<br></p>
<p>技术要求：</p>
<p>1、一年以上 Python 开发经验，掌握面向对象分析和设计，了解设计模式</p>
<p>2、掌握HTTP协议，熟悉MVC、MVVM等概念以及相关WEB开发框架</p>
<p>3、掌握关系数据库开发设计，掌握 SQL，熟练使用 MySQL/PostgreSQL 中的一种<br></p>
<p>4、掌握NoSQL、MQ，熟练使用对应技术解决方案</p>
<p>5、熟悉 Javascript/CSS/HTML5，JQuery、React、Vue.js</p>
<p>&nbsp;<br></p>
<p>加分项：</p>
<p>大数据，数理统计，机器学习，sklearn，高性能，大并发。</p>

        </div>
""" # 这里题要求将所有<>去掉
result5 = re.sub(r"</?\w+>","",s3) # r"</?\w+>"代表去掉所有<>，其中/?代表有反斜杠/的</>可能有，也可能没有
# split根据匹配进行字符串分割
s4 = "itcast:php,python,cpp-java"
result6 = re.split(r":|,|-",s4)
# print(result6)

# 6.贪婪模式(一般情况下你所提取到的内容比你所期望的要多，问题一般是出在了贪婪模式上)
s5 = "This is a number 234-235-22-423"
r=re.match(".+(\d+-\d+-\d+-\d+)",s5)
print(r.group(1))
#从贪婪模式到非贪婪模式只需要加上一个?,不过需要注意的是这里加上一个?并非将所有贪婪模式全部关闭，它只关闭它前面所修饰的那个数量描述符(+)
#?号前关闭的贪婪模式到哪里截止呢？还需要根据后面一个正则条件来决定
r=re.match("(.+?)(\d+-\d+-\d+-\d+)",s5)
print(r.group(2))
